```python
#!/usr/bin/env python3
"""
V01 Automated Pre-Check: Model Output Quality
Generates stratified test set, runs inference, validates output format,
and prepares evaluation packets for domain expert review.
"""

import json
import random
import requests
import sys
from datetime import datetime
from pathlib import Path
from collections import Counter

# === CONFIGURATION ===
ATLAS_API_BASE = "http://localhost:8000/api/v1"  # Staging endpoint
MODEL_ENDPOINT = f"{ATLAS_API_BASE}/predict/classify"
HEALTH_ENDPOINT = f"{ATLAS_API_BASE}/health"
TEST_DATA_PATH = "/atlas-project/data/test/held_out_firs.jsonl"
OUTPUT_DIR = Path("/atlas-project/docs/validation/v01-evidence")
SAMPLE_SIZE = 20  # Per MoU: 20 predictions for expert review
TOTAL_TEST_SIZE = 100  # Full automated test set
ATLAS_CATEGORIES = [
    "violent_crimes", "crimes_against_women_children", "property_crimes",
    "financial_economic_crime", "cyber_crimes", "organised_serious_crimes",
    "public_order_offences", "traffic_negligence", "missing_persons_misc"
]

def check_model_health():
    """Step 1: Verify model endpoint is live and returning expected format."""
    print("[V01-A1] Checking model health endpoint...")
    try:
        resp = requests.get(HEALTH_ENDPOINT, timeout=10)
        assert resp.status_code == 200, f"Health check failed: HTTP {resp.status_code}"
        health = resp.json()
        assert "model_version" in health, "Missing model_version in health response"
        assert "status" in health and health["status"] == "healthy", "Model not healthy"
        print(f"  G£Ù Model healthy | Version: {health['model_version']}")
        return health["model_version"]
    except Exception as e:
        print(f"  G£˘ FAIL: {e}")
        return None

def load_test_data():
    """Step 2: Load held-out test set and validate ground-truth labels."""
    print(f"[V01-A2] Loading test data from {TEST_DATA_PATH}...")
    records = []
    with open(TEST_DATA_PATH, "r", encoding="utf-8") as f:
        for line in f:
            rec = json.loads(line.strip())
            assert "fir_text" in rec, f"Missing fir_text in record {rec.get('fir_id')}"
            assert "ground_truth_category" in rec, f"Missing ground_truth in {rec.get('fir_id')}"
            assert rec["ground_truth_category"] in ATLAS_CATEGORIES, \
                f"Invalid category {rec['ground_truth_category']} in {rec.get('fir_id')}"
            records.append(rec)
    print(f"  G£Ù Loaded {len(records)} test records")
    # Validate category distribution
    dist = Counter(r["ground_truth_category"] for r in records)
    print(f"  Category distribution: {json.dumps(dict(dist), indent=2)}")
    return records

def stratified_sample(records, n=20):
    """Step 3: Create stratified sample ensuring all 9 categories represented."""
    print(f"[V01-A3] Creating stratified sample of {n} records...")
    by_cat = {}
    for r in records:
        by_cat.setdefault(r["ground_truth_category"], []).append(r)
    
    sample = []
    # Ensure at least 1 per category (if available), then fill remaining randomly
    cats_available = [c for c in ATLAS_CATEGORIES if c in by_cat]
    per_cat = max(1, n // len(cats_available))
    
    for cat in cats_available:
        take = min(per_cat, len(by_cat[cat]))
        sample.extend(random.sample(by_cat[cat], take))
    
    # Fill remaining from full pool (excluding already selected)
    selected_ids = {r["fir_id"] for r in sample}
    remaining_pool = [r for r in records if r["fir_id"] not in selected_ids]
    if len(sample) < n and remaining_pool:
        sample.extend(random.sample(remaining_pool, min(n - len(sample), len(remaining_pool))))
    
    random.shuffle(sample)
    print(f"  G£Ù Sampled {len(sample)} records across {len(set(r['ground_truth_category'] for r in sample))} categories")
    return sample[:n]

def run_inference(sample):
    """Step 4: Run model inference on sample and record predictions."""
    print(f"[V01-A4] Running inference on {len(sample)} records...")
    results = []
    for i, rec in enumerate(sample):
        try:
            resp = requests.post(MODEL_ENDPOINT, json={"text": rec["fir_text"]}, timeout=30)
            assert resp.status_code == 200, f"Inference failed for {rec['fir_id']}: HTTP {resp.status_code}"
            pred = resp.json()
            results.append({
                "eval_id": f"V01-{i+1:03d}",
                "fir_id": rec["fir_id"],
                "fir_text_preview": rec["fir_text"][:500],  # Truncate for review form
                "ground_truth": rec["ground_truth_category"],
                "predicted_category": pred.get("predicted_category"),
                "confidence_score": pred.get("confidence", None),
                "inference_time_ms": pred.get("inference_time_ms", None),
                "expert_verdict": None  # To be filled by domain expert
            })
            print(f"  [{i+1}/{len(sample)}] {rec['fir_id']}: predicted={pred.get('predicted_category')} | truth={rec['ground_truth_category']}")
        except Exception as e:
            print(f"  G£˘ Inference error for {rec['fir_id']}: {e}")
            results.append({
                "eval_id": f"V01-{i+1:03d}",
                "fir_id": rec["fir_id"],
                "error": str(e),
                "expert_verdict": "INFERENCE_FAILURE"
            })
    return results

def compute_automated_metrics(results):
    """Step 5: Compute automated accuracy metrics on full test set."""
    print("[V01-A5] Computing automated metrics...")
    correct = sum(1 for r in results if r.get("predicted_category") == r.get("ground_truth"))
    total = len([r for r in results if "error" not in r])
    accuracy = correct / total if total > 0 else 0
    
    errors = len([r for r in results if "error" in r])
    avg_confidence = sum(r.get("confidence_score", 0) for r in results if r.get("confidence_score")) / max(1, total)
    avg_latency = sum(r.get("inference_time_ms", 0) for r in results if r.get("inference_time_ms")) / max(1, total)
    
    metrics = {
        "total_evaluated": len(results),
        "successful_inferences": total,
        "inference_errors": errors,
        "automated_accuracy": round(accuracy, 4),
        "average_confidence": round(avg_confidence, 4),
        "average_latency_ms": round(avg_latency, 2),
        "automated_pass": accuracy >= 0.75  # Pre-check threshold (expert review is the real gate)
    }
    print(f"  Automated accuracy: {accuracy:.1%} ({'PASS' if metrics['automated_pass'] else 'FAIL'} pre-check)")
    return metrics

def generate_expert_evaluation_packet(sample_results, metrics, model_version):
    """Step 6: Generate the evaluation packet for the domain expert."""
    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)
    
    packet = {
        "validation_gate": "V01 G«ˆ Model Output Quality",
        "generated_at": datetime.now().isoformat(),
        "model_version": model_version,
        "automated_pre_check": metrics,
        "instructions_for_expert": (
            "For each of the following FIR text excerpts, the ATLAS system has predicted a "
            "crime category. Please review each prediction and mark it as CORRECT, INCORRECT, "
            "or AMBIGUOUS. A prediction is AMBIGUOUS if the FIR text could reasonably fall into "
            "multiple categories. For INCORRECT predictions, please write the correct category. "
            "You do NOT need to know the ground-truth label G«ˆ evaluate based on your professional "
            "judgment of what the correct classification should be."
        ),
        "evaluation_items": sample_results,
        "pass_threshold": "GÎ— 16 out of 20 marked CORRECT or AMBIGUOUS (80%)",
        "expert_summary_fields": {
            "expert_name": "_______________",
            "expert_designation": "_______________",
            "date_of_review": "_______________",
            "total_correct": "___/20",
            "total_incorrect": "___/20",
            "total_ambiguous": "___/20",
            "overall_verdict": "[PASS / FAIL]",
            "qualitative_comments": "_______________",
            "signature": "_______________"
        }
    }
    
    output_path = OUTPUT_DIR / f"v01-expert-evaluation-packet-{datetime.now().strftime('%Y%m%d')}.json"
    with open(output_path, "w", encoding="utf-8") as f:
        json.dump(packet, f, ensure_ascii=False, indent=2)
    print(f"  G£Ù Expert evaluation packet written to {output_path}")
    return output_path

def main():
    print("=" * 70)
    print("ATLAS V01 G«ˆ MODEL OUTPUT QUALITY G«ˆ AUTOMATED PRE-CHECK")
    print("=" * 70)
    
    # Step 1: Health check
    model_version = check_model_health()
    if not model_version:
        print("\nG£˘ GATE BLOCKED: Model endpoint not available. Cannot proceed.")
        sys.exit(1)
    
    # Step 2: Load data
    records = load_test_data()
    assert len(records) >= TOTAL_TEST_SIZE, f"Insufficient test data: {len(records)} < {TOTAL_TEST_SIZE}"
    
    # Step 3: Stratified sample for expert review
    expert_sample = stratified_sample(records, SAMPLE_SIZE)
    
    # Step 4: Run inference on expert sample
    sample_results = run_inference(expert_sample)
    
    # Step 5: Compute automated metrics (on full test set for completeness)
    full_results = run_inference(records)
    metrics = compute_automated_metrics(full_results)
    
    # Step 6: Generate expert packet
    packet_path = generate_expert_evaluation_packet(sample_results, metrics, model_version)
    
    # Final pre-check verdict
    print("\n" + "=" * 70)
    print("V01 AUTOMATED PRE-CHECK SUMMARY")
    print("=" * 70)
    print(f"  Model version:      {model_version}")
    print(f"  Test set size:      {len(records)}")
    print(f"  Expert sample size: {len(expert_sample)}")
    print(f"  Automated accuracy: {metrics['automated_accuracy']:.1%}")
    print(f"  Avg latency:        {metrics['average_latency_ms']:.0f} ms")
    print(f"  Pre-check verdict:  {'G£Ù PASS' if metrics['automated_pass'] else 'G£˘ FAIL'}")
    print(f"  Expert packet at:   {packet_path}")
    print(f"\n  GÂ∆ Next step: Deliver packet to domain expert for independent review.")
    print("=" * 70)

if __name__ == "__main__":
    main()
```

#### Phase B: Domain Expert Independent Review (Human Protocol)

**Step B1:** Test administrator (Aditya) prints or prepares digital copies of the expert evaluation packet generated by Phase A.

**Step B2:** Test administrator delivers the packet to the domain expert with the following verbal instructions:

> "You have 20 FIR text excerpts, each with a predicted crime category from our AI system. For each prediction, please mark it CORRECT, INCORRECT, or AMBIGUOUS based on your professional judgment. You do not need to know what the system was 'supposed' to predict G«ˆ your independent assessment of the correct category is the ground truth for this review. If you mark INCORRECT, please write what the correct category should be. There is no time limit. Please ask me to leave the room if my presence influences your review."

**Step B3:** Test administrator leaves the room or mutes/disconnects from the call.

**Step B4:** Domain expert completes the evaluation form and signs it.

**Step B5:** Test administrator collects the completed form and computes:
- Count of CORRECT verdicts
- Count of INCORRECT verdicts  
- Count of AMBIGUOUS verdicts
- Pass calculation: (CORRECT + AMBIGUOUS) / 20 GÎ— 80%

**Step B6:** Test administrator records results in the verdict document (does NOT alter expert's markings).

```python
#!/usr/bin/env python3
"""
V02 Automated Bias Audit
Computes fairness metrics across all demographic axes for all 6 pilot districts.
Outputs a structured report for independent reviewer adjudication.
"""

import json
import pandas as pd
import numpy as np
from datetime import datetime
from pathlib import Path
from scipy import stats

# Attempt Fairlearn import; provide fallback manual calculation
try:
    from fairlearn.metrics import (
        demographic_parity_difference,
        equalized_odds_difference,
        MetricFrame
    )
    from sklearn.metrics import accuracy_score, f1_score, precision_score, recall_score
    FAIRLEARN_AVAILABLE = True
except ImportError:
    FAIRLEARN_AVAILABLE = False
    from sklearn.metrics import accuracy_score, f1_score, precision_score, recall_score

# === CONFIGURATION ===
PREDICTIONS_PATH = "/atlas-project/data/test/v01_full_predictions.jsonl"
DEMOGRAPHICS_PATH = "/atlas-project/data/test/demographic_annotations.jsonl"
OUTPUT_DIR = Path("/atlas-project/docs/validation/v02-evidence")
THRESHOLDS = {  # From ADR-D05
    "demographic_parity_difference": 0.05,
    "district_f1_cv": 0.15,         # Coefficient of variation across districts
    "gender_recall_gap": 0.10,       # For Ch.V offences
    "language_f1_gap": 0.08,         # Gujarati vs English
    "caste_accuracy_gap": 0.05       # Across caste categories
}

PILOT_DISTRICTS = [
    "Ahmedabad_City", "Ahmedabad_Rural", "Surat_City",
    "Vadodara_City", "Rajkot_City", "Gandhinagar"
]

DEMOGRAPHIC_AXES = [
    "district", "victim_gender", "victim_caste_category",
    "primary_language", "offence_chapter"
]

def load_data():
    """Load predictions merged with demographic annotations."""
    print("[V02-A1] Loading prediction and demographic data...")
    preds = pd.read_json(PREDICTIONS_PATH, lines=True)
    demos = pd.read_json(DEMOGRAPHICS_PATH, lines=True)
    df = preds.merge(demos, on="fir_id", how="inner")
    df["is_correct"] = (df["predicted_category"] == df["ground_truth_category"]).astype(int)
    print(f"  G£Ù Loaded {len(df)} records with demographics")
    print(f"  Districts: {df['district'].nunique()}")
    print(f"  Gender distribution: {df['victim_gender'].value_counts().to_dict()}")
    print(f"  Caste distribution: {df['victim_caste_category'].value_counts().to_dict()}")
    return df

def compute_axis_metrics(df, axis_col, axis_label):
    """Compute accuracy, F1, and disparity metrics for one demographic axis."""
    print(f"\n[V02] Analysing axis: {axis_label} ({axis_col})...")
    groups = df[axis_col].unique()
    results = {}
    
    for group in groups:
        mask = df[axis_col] == group
        subset = df[mask]
        if len(subset) < 5:
            print(f"  GÐ· Skipping {group}: only {len(subset)} records (< 5 minimum)")
            continue
        acc = accuracy_score(subset["ground_truth_category"], subset["predicted_category"])
        f1 = f1_score(subset["ground_truth_category"], subset["predicted_category"],
                      average="weighted", zero_division=0)
        results[group] = {
            "n": len(subset),
            "accuracy": round(acc, 4),
            "f1_weighted": round(f1, 4)
        }
        print(f"  {group}: n={len(subset)}, accuracy={acc:.3f}, F1={f1:.3f}")
    
    # Compute disparity
    accuracies = [v["accuracy"] for v in results.values()]
    f1s = [v["f1_weighted"] for v in results.values()]
    
    disparity = {
        "axis": axis_label,
```

```python
#!/usr/bin/env python3
"""
V05 Automated Multilingual Quality Checks
Validates Unicode rendering, script detection accuracy, and encoding integrity.
"""

import json, re, sys
from pathlib import Path
from datetime import datetime
from collections import Counter

OUTPUT_DIR = Path("/atlas-project/docs/validation/v05-evidence")

# Gujarati Unicode range
GUJARATI_RANGE = range(0x0A80, 0x0B00)

def is_gujarati_char(ch):
    return ord(ch) in GUJARATI_RANGE

def analyse_script_distribution(text):
    """Analyse character-level script distribution."""
    gujarati = sum(1 for ch in text if is_gujarati_char(ch))
    latin = sum(1 for ch in text if ch.isascii() and ch.isalpha())
    devanagari = sum(1 for ch in text if 0x0900 <= ord(ch) <= 0x097F)
    numeric = sum(1 for ch in text if ch.isdigit())
    other = len(text) - gujarati - latin - devanagari - numeric - text.count(' ')
    total_alpha = gujarati + latin + devanagari
    return {
        "gujarati_chars": gujarati,
        "latin_chars": latin,
        "devanagari_chars": devanagari,
        "numeric_chars": numeric,
        "other_chars": other,
        "gujarati_ratio": round(gujarati / max(1, total_alpha), 3),
        "dominant_script": (
            "gujarati" if gujarati > latin and gujarati > devanagari
            else "latin" if latin > gujarati and latin > devanagari
            else "devanagari" if devanagari > gujarati and devanagari > latin
            else "mixed"
        )
    }

def check_encoding_integrity(records):
    """Check for mojibake, replacement characters, and encoding errors."""
    print("[V05-A1] Checking encoding integrity...")
    issues = []
    for rec in records:
        text = rec.get("fir_text", "")
        fir_id = rec.get("fir_id", "unknown")
        # Check for replacement character
        if '\ufffd' in text:
            issues.append({"fir_id": fir_id, "issue": "Replacement character (U+FFFD) found G«ˆ encoding corruption"})
        # Check for common mojibake patterns
        if re.search(r'[+‚-Ò+‚-¨+‚-++È]', text):
            issues.append({"fir_id": fir_id, "issue": "Possible UTF-8/Latin-1 mojibake detected"})
        # Check for null bytes
        if '\x00' in text:
            issues.append({"fir_id": fir_id, "issue": "Null byte in text"})
    
    print(f"  Encoding issues found: {len(issues)}")
    return issues

def check_classification_by_language(records):
    """Verify that classification accuracy doesn't vary by dominant script."""
    print("[V05-A2] Checking classification accuracy by dominant script...")
    by_script = {}
    for rec in records:
        dist = analyse_script_distribution(rec["fir_text"])
        script = dist["dominant_script"]
        correct = rec.get("predicted_category") == rec.get("ground_truth_category")
        by_script.setdefault(script, {"correct": 0, "total": 0})
        by_script[script]["total"] += 1
        if correct:
            by_script[script]["correct"] += 1
    
    results = {}
    for script, counts in by_script.items():
        acc = counts["correct"] / max(1, counts["total"])
        results[script] = {"n": counts["total"], "accuracy": round(acc, 4)}
        print(f"  {script}: n={counts['total']}, accuracy={acc:.3f}")
    
    # Check for language performance gap (ADR-D05 threshold: 0.08)
    accuracies = [v["accuracy"] for v in results.values() if v["n"] >= 5]
    gap = max(accuracies) - min(accuracies) if len(accuracies) >= 2 else 0
    print(f"  Max accuracy gap: {gap:.4f} (threshold GÎÒ 0.08)")
    
    return results, gap
```

## AUTOMATED GATE AGGREGATOR

```python
#!/usr/bin/env python3
"""
ATLAS Validation Gate Aggregator
Reads all verdict documents and produces a single deployment readiness report.
"""

import json
from pathlib import Path
from datetime import datetime

VALIDATION_DIR = Path("/atlas-project/docs/validation")

def aggregate_gates():
    gates = {}
    for i in range(1, 13):
        gate_id = f"V{i:02d}"
        pattern = f"{gate_id}-*-verdict-*.md"
        files = list(VALIDATION_DIR.glob(pattern))
        if files:
            latest = sorted(files)[-1]  # Most recent verdict
            content = latest.read_text()
            # Simple extraction G«ˆ in production, use structured YAML frontmatter
            if "**PASS**" in content or "PASS" in content.split('\n')[0:5]:
                verdict = "PASS"
            elif "CONDITIONAL" in content:
                verdict = "CONDITIONAL PASS"
            elif "FAIL" in content:
                verdict = "FAIL"
            else:
                verdict = "UNKNOWN"
            gates[gate_id] = {"verdict": verdict, "file": str(latest), "date": latest.stem.split("-")[-1]}
        else:
            gates[gate_id] = {"verdict": "NOT EXECUTED", "file": None, "date": None}
    
    # Determine deployment readiness
    verdicts = [g["verdict"] for g in gates.values()]
    if all(v in ("PASS", "CONDITIONAL PASS") for v in verdicts):
        readiness = "DEPLOYMENT READY"
    elif "NOT EXECUTED" in verdicts:
        readiness = "INCOMPLETE G«ˆ gates pending"
    else:
        failing = [gid for gid, g in gates.items() if g["verdict"] == "FAIL"]
        readiness = f"BLOCKED G«ˆ failing gates: {', '.join(failing)}"
    
    report = {
        "timestamp": datetime.now().isoformat(),
        "gates": gates,
        "deployment_readiness": readiness,
        "total_gates": 12,
        "passed": sum(1 for v in verdicts if v == "PASS"),
        "conditional": sum(1 for v in verdicts if v == "CONDITIONAL PASS"),
        "failed": sum(1 for v in verdicts if v == "FAIL"),
        "pending": sum(1 for v in verdicts if v == "NOT EXECUTED")
    }
    
    print("=" * 60)
    print("ATLAS DEPLOYMENT READINESS REPORT")
    print("=" * 60)
    for gid, g in sorted(gates.items()):
        icon = {"PASS": "G£Ù", "CONDITIONAL PASS": "GÐ·", "FAIL": "G£˘", "NOT EXECUTED": "G˘Ô"}.get(g["verdict"], "?")
        print(f"  {icon} {gid}: {g['verdict']}")
    print(f"\n  READINESS: {readiness}")
    print("=" * 60)
    
    output = VALIDATION_DIR / f"deployment-readiness-{datetime.now().strftime('%Y%m%d')}.json"
    with open(output, "w") as f:
        json.dump(report, f, indent=2)
    print(f"  Report: {output}")

if __name__ == "__main__":
    aggregate_gates()
```
